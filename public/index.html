#!/usr/bin/env python3 """ Rename Phantum -> Your Situation in Tacorari index.html Run from folder containing index.html: python3 patch_tacorari.py """ import os INPUT = 'index.html' if not os.path.exists(INPUT): print(f"ERROR: {INPUT} not found.") print("Put this script in the same folder as your Tacorari index.html") exit(1) html = open(INPUT, 'r', encoding='utf-8').read() # Replace the card name (bold display name) html = html.replace( 'Phantum', 'Your Situation' ) print(" Renamed: card name") # Replace HTML comment (cosmetic, but consistent) html = html.replace('', '') print(" Renamed: HTML comment") # Privacy policy text - rename the TOOL name but keep it readable html = html.replace( '(including Phantum) collect', '(including Your Situation) collect' ) html = html.replace( 'When you use Phantum,', 'When you use Your Situation,' ) html = html.replace( 'Tacorari does not use cookies. Phantum uses', 'Tacorari does not use cookies. Your Situation uses' ) print(" Renamed: privacy policy references") # Note: We keep the URL /phantum unchanged because that's the route path # Note: We keep tacorari.eu/phantum as the displayed link text unchanged for same reason open(INPUT, 'w', encoding='utf-8').write(html) print(f"\nDone. File size: {len(html)} bytes")
